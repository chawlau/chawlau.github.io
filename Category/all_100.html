归并排序和快速排序

归并排序

void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
   for(int i=m-1, j=n-1, k=m+n-1; j>=0; )
       nums1[k--] = i>=0 && nums1[i]>nums2[j]? nums1[i--]: nums2[j--];
}
