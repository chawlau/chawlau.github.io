namedtuple

from collections import namedtuple

User = namedtuple('User', ['name', 'age', 'height', 'edu'])
user = User(name = 'bobby', age = 29, height=175, edu='master')

user_tuple = ('bobby', 29, 175, 'super master')
user_list = ('bobby', 29, 175)
user = User(*user_tuple)
user_dict = {
    'name': 'bobby',
    'age': 29,
    'height': 175,
    'edu': 'pre master'
}
user = User(**user_dict)
# user = User._make(user_list, 'pre master')

# namedtuple 直接转换成dict
user_info_dict = user._asdict()

#拆包功能
name, age, *other = user

defaultdict
from collections import defaultdict
user_dict = {}
users = ['bobby1', 'bobby2', 'bobby3']
for user in users:
    if user not in user_dict:
        user_dict[user] = 1
    else:
        user_dict[user] += 1
user_dict

for user in users:
    user_dict.setdefault(user, 0)
    user_dict[user] += 1
user_dict

user_dict = defaultdict(int)
for user in users:
    user_dict[user] += 1
user_dict

def gen_default():
    return {
        "name": "",
        "nums": 0
    }

user_dict = defaultdict(gen_default)
user_dict['group1']
user_dict

from collections import deque

user_deque = deque(('bobby1', 'bobby2'))
user_deque = deque(['bobby1', 'bobby2', 'bobby3'])
user_deque.appendleft('9')
user_deque.append('10')
user_deque.insert(0, 'bobby3')
user_deque
user_deque2 = user_deque.copy()
user_deque2.appendleft('7')

Counter

from collections import Counter
users = ['bobby1', 'bobby2', 'bobby3', 'bobby4', 'bobby5', 'bobby6',]
users2 = ['bobby1', 'bobby2', 'bobby1', 'bobby2', 'bobby1', 'bobby2',]
user_counter = Counter(users)
user_counter.update(users2)
# 统计前N个字符串的
ret = user_counter.most_common(2)
user_counter
ret

ChainMap

from collections import ChainMap

user_dict1 = {'a':'bobby1', 'b':'bobby2'}
user_dict2 = {'c':'bobby1', 'd':'bobby2'}
new_dict = ChainMap(user_dict1, user_dict2)
new_dict.new_child({'aa':'aa', 'bb':'bb'})

#如果有重复元素，就不会再出现第二次
for k, v in new_dict.items():
    print(k, v)
new_dict.maps
new_dict.maps[0]['a'] = 'bobby'
