  • 一个函数和其相关的引用的环境组合而成的实体
func Adder() func(int) int {
  var x int
  return func(delta int) int {
    x += delta
    return x
  }
}

func makeSuffixFunc(suffix string) func(string) string {
  return func(name string) string {
    if !strings.HasSuffix(name, suffix) {
      return name + suffix
    }
    return name
  }
}

func makeSlice(s slice, cap int) slice {
  s.ptr = new([100]int)
  s.cap = cap
  s.len = 0
  return s
 }

func modify(s slice) {
  s.ptr[1] = 1000
}

func testSlice2() {
  var s1 slice
  s1 = makeSlice(s1, 10)
  s1.ptr[0] = 100
  modify(s1)
  fmt.Println(s1.ptr)
}

func TestMapForSet(t *testing.T) {
	mySet := map[int]bool{}
	mySet[1] = true
	n := 3
	if mySet[n] {
		t.Logf("%d is existing", n)
	} else {
		t.Logf("%d is not existing", n)
	}
	mySet[3] = true
	t.Log(len(mySet))
	delete(mySet, 1)
	n = 1
	if mySet[n] {
		t.Logf("%d is existing", n)
	} else {
		t.Logf("%d is not existing", n)
	}
}
