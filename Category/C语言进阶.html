//version1
#define MAX(x,y) ((x) > (y) ? (x) : (y))

//version2
#define MAX(x,y)({     \
    int _x = x;        \
    int _y = y;        \
    _x > _y ? _x : _y; \
})

//versron3
#define MAX(type,x,y)({     \
    type _x = x;        \
    type _y = y;        \
    _x > _y ? _x : _y; \
})

//内核中实现
#define min_t(type, x, y) ({            \
    type __min1 = (x);          \
    type __min2 = (y);          \
    __min1 < __min2 ? __min1 : __min2; })

#define max_t(type, x, y) ({            \
    type __max1 = (x);          \
    type __max2 = (y);          \
    __max1 > __max2 ? __max1 : __max2; })

container_of

#define offsetof(TYPE, MEMBER) ((size_t) &((TYPE *)0)->MEMBER)
#define  container_of(ptr, type, member) ({    \
     const typeof( ((type *)0)->member ) *__mptr = (ptr); \
     (type *)( (char *)__mptr - offsetof(type,member) );})
     
struct student
{
    int age;
    int num;
    int math;
};
int main(void)
{
    struct student stu = { 20, 1001, 99};

    int *p = &stu.math;
    struct student *stup = NULL;
    stup = container_of( p, struct student, math);
    printf("%p\n",stup);
    printf("age: %d\n",stup->age);
    printf("num: %d\n",stup->num);

    return 0;     
}

范围赋值

  • 通过数组索引，我们可以直接给指定的数组元素赋值。除此之外，一个结构体变量的初始化，也可以通过指定某个结构体域直接赋值。
#include <stdlib.h>
#include <stdio.h>


int main(void)
{
  int b[100] = { [10 ... 30] = 1, [50 ... 60] = 2 };
  for(int i = 0; i < 100; i++)
  {
    printf("%d  ", b[i]);
    if( i % 10 == 0)
      printf("\n");
  }

  int j = 5;
  switch(j)
  {
    case 1:
      printf("1\n");
      break;
    case 2 ... 8:
      printf("%d\n",j);
      break;
    case 9:
      printf("9\n");
      break;
    default:
      printf("default!\n");
      break;
  }

  return 0;
}

数据类型

内存

  • 内存是线性的，按字节为单位进行编址的

补码

  • 0的补码是0
  • 符号位和普通位一并参与运算
  • 补码可以使加减乘除，全部变加法

数据类型

  • 数据类型提供了申请内存单元的大小和访问规则，数据类型是假设在线性内存上的一种逻辑关系
  • unsigned扩充前面应该补0
  • -1转换成补码存在计算机中
