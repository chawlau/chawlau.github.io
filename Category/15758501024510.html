可重入的锁

  • 同一个线程里面，可以连续调用多次acquire,一定要注意acquire的次数
  • RLock可重入锁
from threading import Lock, RLock, Condition

total = 0
lock = RLock()

def add():
    global lock
    global total
    for i in range(10000):
        lock.acquire()
        lock.acquire()
        total += 1
        lock.release()
        lock.release()

def desc():
    global total
    global lock
    for i in range(10000):
        # lock.acquire()
        lock.acquire()
        total -= 1
        lock.release()
        # lock.release()

import threading

thread1 = threading.Thread(target=add)
thread2 = threading.Thread(target=desc)

thread1.start()
thread2.start()
thread2.join()
thread2.join()
print("total {}".format(total))

协同

import threading

class XiaoAi(threading.Thread):
    def __init__(self, lock):
        super().__init__(name="xiaoai")
        self.lock = lock

    def run(self):
        self.lock.acquire()
        print("{}. at ".format(self.name))
        self.lock.release()

        self.lock.acquire()
        print("{}. ok ".format(self.name))
        self.lock.release()

class TianMao(threading.Thread):
    def __init__(self, lock):
        super().__init__(name="tiaomao")
        self.lock = lock

    def run(self):
        self.lock.acquire()
        print("{} : xiaoai roomate".format(self.name))
        self.lock.release()

        self.lock.acquire()
        print("{} : let's talk".format(self.name))
        self.lock.release()

if __name__ == '__main__':
    lock = threading.Lock()
    thread1 = XiaoAi(lock)
    thread2 = TianMao(lock)
    thread1.start()
    thread2.start()
    thread1.join()
    thread2.join()
  • condition启动顺序很重要,在调用with cond之后才能调用wait或者notify方法
  • condition有两层锁， 一把底层锁会在线程调用了wait方法的时候释放， 上面的锁会在每次调用wait的时候分配一把并放入到cond的等待队列中，等到notify方法的唤醒
    xiaoai.start()
import threading

class XiaoAi(threading.Thread):
    def __init__(self, cond):
        super().__init__(name="xiaoai")
        self.cond = cond

    def run(self):
        with self.cond:
            self.cond.wait()
            print('{}: 在 '.format(self.name))
            self.cond.notify()

            self.cond.wait()
            print('{}: 好阿 '.format(self.name))
            self.cond.notify()

            self.cond.wait()
            print('{}: 君住长江尾 '.format(self.name))
            self.cond.notify()

            self.cond.wait()
            print('{}: 共饮长江水 '.format(self.name))
            self.cond.notify()

            self.cond.wait()
            print('{}: 此恨何时已 '.format(self.name))
            self.cond.notify()

            self.cond.wait()
            print('{}: 定不负相思意 '.format(self.name))
            self.cond.notify()

class TianMao(threading.Thread):
    def __init__(self, cond):
        super().__init__(name="天猫精灵")
        self.cond = cond

    def run(self):
        with self.cond:
            print('{}: 小爱同学 '.format(self.name))
            self.cond.notify()
            self.cond.wait()

            print('{}: 我们来对古诗吧 '.format(self.name))
            self.cond.notify()
            self.cond.wait()

            print('{}: 我住长江头 '.format(self.name))
            self.cond.notify()
            self.cond.wait()

            print('{}: 日日思君不见君 '.format(self.name))
            self.cond.notify()
            self.cond.wait()

            print('{}: 此水几时休 '.format(self.name))
            self.cond.notify()
            self.cond.wait()

            print('{}: 只愿君心似我心 '.format(self.name))
            self.cond.notify()
            self.cond.wait()

if __name__ == '__main__':
    from concurrent import futures
    cond = threading.Condition()

    tianmao = TianMao(cond)

    xiaoai.start()

ThreadPoolExcutor

from concurrent.futures import ThreadPoolExecutor, as_completed, wait, FIRST_COMPLETED
from concurrent.futures import Future
from multiprocessing import Pool


import time

def get_html(times):
    time.sleep(times)
    print('get page {} process'.format(times))
    return times

executor = ThreadPoolExecutor(max_workers=2)
执行任务获取结果
task1 = executor.submit(get_html, (3))
task2 = executor.submit(get_html, (2))
#
print(task1.result())
print(task2.cancel()) //取消结果
多任务
urls = [3, 2, 4]

all_task = [executor.submit(get_html, (url)) for url in urls]

for future in as_completed(all_task):
    data = future.result()
    print('get {} page'.format(data))


#通过excutor获取已经完成的task
for data in executor.map(get_html, urls):
    print('get {} page'.format(data))
wait获取结果
#wait
wait(all_task, return_when=FIRST_COMPLETED)
print('main')
